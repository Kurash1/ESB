new class event_modifier = {
	id = `{this:id}`
	list = event_modifiers
	default = {
		name = "{this:id}"
	}
	args = {
		name = string
		desc = string?
		modifier = named_modifier
	}
	simple_transpile = {
		write_file "common/event_modifiers/arc.txt" = {
			foreach $modifier in event_modifiers {
				new localisation `{$modifier}` = "{$modifier:name}"
				when [exists = $modifier:desc] {
					new localisation `desc_{$modifier}` = "{$modifier:desc}"
				}
				$modifier = {
					$modifier:modifier
				}
			}
		}
	}
}
new class historical_war = {
	list = historical_wars
	args = {
		name = string
		war_goal = named_block
		dates = named_effect
	}
	simple_transpile = {
		foreach $war in historical_wars {
			write_file "history/wars/{$war}.txt" = {
				name = `"{$war:name}"`
				war_goal = {
					$war:war_goal
				}
				$war:dates
			}
		}
	}
}

new class modifier_definition = {
	id = `{this:id}`
	list = modifier_definitions
	default = {
		is_percentage = no
		precision = 1
		trigger = {
			always = yes
		}
	}
	args = {
		name = string
		is_percentage = bool
		trigger = named_trigger
		base_value = string?
		precision = int
	}
	on_create = {
		when [this:is_percentage == yes] {
			DEFINE_MODIFIER = {
				key = `{this:id}`
				text = `{this:name}`
				localisation_key = `{this:id}`
				multiplier = 100
				percent = yes
				is_good = true
				is_bool = false
				is_multiplicative = true
				precision = this:precision
			}

		}
		when [this:is_percentage == no] {
			DEFINE_MODIFIER = {
				key = `{this:id}`
				text = `{this:name}`
				localisation_key = `{this:id}`
				multiplier = 1
				percent = no
				is_good = true
				is_bool = false
				is_multiplicative = true
				precision = this:precision
			}
		}
	}
}

new class script_variable = {
	list = variables
	default = {
		tag = ""
	}
	args = {
		initial_value = float?
		calculate = named_effect?
		tag = string
		name = string
	}
	simple_transpile = {
		foreach $ in variables {
			new localisation `{$:id}` = `{$:name}`
		}
	}
	init = {
		args = bool
		transpile = {
			set_variable = {
				which = this:id
				value = this:initial_value
			}
		}
	}
}
