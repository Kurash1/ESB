#Temple of Miraak
country_event = {
	id = es_miraak.1
	title = es_miraak.1.t
	desc = es_miraak.1.d
	picture = ES_TEMPLE_OF_MIRAAK_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			set_variable = {
				which = miraak_cult_favour
				value = 0
			}
		}
	}
	
	#Let them be
	option = {
		name = es_miraak.1.a
		3314 = {
			add_base_tax = 3
			add_base_production = 3
			add_base_manpower = 3
			
			change_religion = miraak_cult
			add_province_modifier = {
				name = miraak_cult_recently_converted
				days = 365
			}
		}
		
		hidden_effect = {
			country_event = {
				id = es_miraak.2
				days = 182
				random = 366
			}
			change_variable = {
				which = miraak_cult_favour
				value = 1
			}
		}
	}
	
	#We cannot allow cultists to live on our land
	option = {
		name = es_miraak.1.b
		3314 = {
			miraak_cult_rebels = 1
		}
		
		hidden_effect = {
			country_event = {
				id = es_miraak.2
				days = 365
				random = 730
			}	
		}
	}
}
