province_event = {
	id = es_esb.1
	title = es_esb.1.t
	desc = es_esb.1.d
	picture = TRADEGOODS_eventPicture
	
	is_triggered_only = yes
	
	#Nothing
	option = {
		name = es_esb.0.a
	}
	
	#Ebony
	option = {
		name = es_esb.1.a
		trigger = {
			OR = {
				has_terrain = hills 
				has_terrain = mountain
				has_terrain = highlands 
				has_terrain = deadlands_boiling_coast
				has_terrain = deadlands_burning_hills
				
				has_terrain = dwemer_stronghold
				has_terrain = deadlands_daedric_keep
				has_terrain = deadlands_volcanic_peak
				has_terrain = deadlands_lava_cave
				
				has_terrain = ashlands 
				has_terrain = lava
			}
		}
		
		if = {
			limit = {
				OR = {
					has_terrain = hills 
					has_terrain = mountain
					has_terrain = highlands 
					has_terrain = deadlands_boiling_coast
					has_terrain = deadlands_burning_hills
				}
			}
			
			# 25 x Base Price x Terrain modifier (50% to 200%)
			# 25 x 8.0 x 2.0 = 400
			
			adm_power_cost  = 400
			dip_power_cost  = 400
			mil_power_cost  = 400
			
			change_trade_goods = ebony
		}
		
		if = {
			limit = {
				OR = {
					has_terrain = dwemer_stronghold
					has_terrain = deadlands_daedric_keep
					has_terrain = deadlands_volcanic_peak
					has_terrain = deadlands_lava_cave
				}
			}
			
			# 25 x Base Price x Terrain modifier (50% to 200%)
			# 25 x 8.0 x 1.0 = 200
			
			adm_power_cost  = 200
			dip_power_cost  = 200
			mil_power_cost  = 200
			
			change_trade_goods = ebony
		}
		
		if = {
			limit = {
				OR = {
					has_terrain = ashlands 
					has_terrain = lava
				}
			}
			
			# 25 x Base Price x Terrain modifier (50% to 200%)
			# 25 x 8.0 x 0.5 = 100
			
			adm_power_cost  = 100
			dip_power_cost  = 100
			mil_power_cost  = 100
			
			change_trade_goods = ebony
		}
	}
}
