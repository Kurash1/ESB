federal_consolidation = {
	gfx = GFX_federal_consolidation
	can_pick = {
		num_federation_advancements = 16
		federation_size = 5
	}

	effect = {
		# Unite federation
		custom_tooltip = INTEGRATE_FEDERATION_MEMBERS
		hidden_effect = {
			every_tribal_land_province = {
				limit = { NOT = { owner = { exists = yes } } }
				settle_province = ROOT
			}
			every_federation_member = {
				limit = {
					NOT = { tag = PREV }
					ai = yes
				}
				ROOT = {
					inherit = PREV
				}
			}
		}
		
		if = {
			limit = {
				NOT = { government_rank = 3 }
			}
			set_government_rank_force = 3
		}
		add_country_modifier = {
			name = "centralization_modifier"
			duration = 7300
		}
		add_prestige = 25
		
		form_federation_nation = {
			post_effect = {
				add_permanent_claims_for_federation_nation = { ORIGINAL = ROOT TAG = THIS }
				swap_federation_idea_group = yes
			}
		}

		hidden_effect = {
			disband_federation = yes
		}
	}

	ai_chance = {
		factor = 1
	}
}
