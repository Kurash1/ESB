new named_effect survey_calculate_expenses = {
	variables:survey_monthly_expenses:set = 0
	
	#Administrative Costs
	variables:survey_administrative_costs:set_var = survey_manpower
	variables:survey_administrative_costs:divide = 1000
	variables:survey_administrative_costs:multiply = SURVEY_SCOUTS_ADMINISTRATIVE_COST
	variables:survey_monthly_expenses:add_var = survey_administrative_costs
	
	#Salaries
	variables:survey_salary:set_var = survey_manpower
	variables:survey_salary:divide = 1000
	variables:survey_salary:multiply = SURVEY_SALARY_PER_STACK
	variables:survey_monthly_expenses:add_var = survey_salary
	
	#Maintenance
	variables:survey_scout_maintenance:set_var = survey_manpower
	variables:survey_scout_maintenance:divide = 1000
	variables:survey_scout_maintenance:multiply = SURVEY_MAINTENANCE_PER_STACK
	variables:survey_monthly_expenses:add_var = survey_scout_maintenance
	
}
