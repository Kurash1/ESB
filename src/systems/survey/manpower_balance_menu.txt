new customizable_localization GetSurveyManpowerSum = {
	{
		text = "§R[This.survey_manpower_sum.GetValue]§!"
		trigger = {
			NOT = {
				check_variable = {
					which = survey_manpower_sum
					value = 0
				}
			}
		}
	}
	{
		text = "§G+[This.survey_manpower_sum.GetValue]§!"
		trigger = {
			check_variable = {
				which = survey_manpower_sum
				value = 0.01
			}
		}
	}
	{
		text = "§Y[This.survey_manpower_sum.GetValue]§!"
		trigger = {
			always = yes
		}
	}
}
