new named_effect survey_calculate_expenses = {
	variables:survey_monthly_expenses:set = 0
	
	#Administrative Costs
	#
	# Stacks * SURVEY_SCOUTS_ADMINISTRATIVE_COST_MANPOWER_MULT
	# + SURVEY_SCOUTS_BASE_ADMINISTRATIVE_COST
	# * modifier:scouts_administrative_cost_modifier
	#
	&survey_administrative_costs := survey_monthly_recruits_base
	&survey_administrative_costs *= SURVEY_SCOUTS_ADMINISTRATIVE_COST_PER_STACK
	
	&survey_temp &= modifier:scouts_administrative_cost
	&survey_administrative_costs += survey_temp
	
	&survey_temp &= modifier:scouts_administrative_cost_modifier
	&survey_administrative_costs *= survey_temp
	
	&survey_monthly_expenses += survey_administrative_costs
	
	#Maintenance
	#
	# Stacks * SURVEY_MAINTENANCE_PER_STACK * modifier:scouts_maintenance_cost_modifier
	#
	&survey_scout_maintenance := survey_manpower
	&survey_scout_maintenance *= SURVEY_MAINTENANCE_PER_STACK
	
	&survey_temp &= modifier:scouts_maintenance_cost_modifier
	&survey_scout_maintenance *= survey_temp
	
	&survey_monthly_expenses += survey_scout_maintenance
}
