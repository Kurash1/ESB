new modifier_definition global_expedition_progess = {
	name = "Global Monthly Expedition Progress"
	is_percentage = no
	trigger = {
		always = yes
	}
}
new modifier monthly_expedition_progress = {
	args = {
		amount = float
		trigger = trigger
	}
	transpile = {
		`expedition_progess = {args:amount}`
		
		expedition_progress_pulse +=! {
			if [args:trigger] {
				cause_expedition_progress = args:amount
			}
		}
	}
}

	name = "Expedition Camp"
	modifier = {

			amount = 1

		}
