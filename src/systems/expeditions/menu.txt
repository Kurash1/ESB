new custom_button start_expedition = {
	tooltip = "Start Expedition"
	potential = {
		has_province_flag = expedition_target
	}
	trigger = {
		owned_by = FROM
	}
	effect = {
		if [has_province_modifier = expedition_camp] {
			remove_province_modifier = expedition_camp
			remove_province_modifier = expedition_mod_cash_mod
			remove_province_modifier = expedition_mod_camp_mod
			remove_province_modifier = expedition_mod_explore_mod
			remove_province_modifier = expedition_mod_merc_mod
		}
		else {
			add_permanent_province_modifier = {
				name = expedition_camp
				duration = -1
			}
		}
	}
}
