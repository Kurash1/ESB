	args = {
		level = int
		offset = int
	}
	transpile = {
		new localisation `adm_tech_cs_{(args:level*3+args:offset)}_name` = `Available at §Y{(2568+args:level*50)}§!`
		new localisation `dip_tech_cs_{(args:level*3+args:offset)}_name` = `Available at §Y{(2568+args:level*50)}§!`
		new localisation `mil_tech_cs_{(args:level*3+args:offset)}_name` = `Available at §Y{(2568+args:level*50)}§!`
		
		new localisation `adm_tech_cs_{(args:level*3+args:offset)}_desc` = ""
		new localisation `dip_tech_cs_{(args:level*3+args:offset)}_desc` = ""
		new localisation `mil_tech_cs_{(args:level*3+args:offset)}_desc` = ""
	}
}
