new country_event theocracy.2 = {
	title = "The Next Sybil has been found!"
	desc = "As the position of the next Sybil has been empty for some time, our sisters have performed the 'Exalted Protocol of the Dibellan Sybil' which has revealed to us the home of the next Sybil. [es_theocracy_province.GetName] is where the child lives."
	picture = ES_THEOCRACY_HEIR_ELECTIONS_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_reform = tier_1_sybildom
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = in_theocracy_heir_selection
			
			random_owned_province = {
				save_event_target_as = es_theocracy_province
			}
		}
	}
	
	options = {
		{
			name = "Great!"
			effect = {
				define_heir = {
					age = 10
					dynasty = "Nymph-Dance"
					claim = 100
					hidden = yes
					culture = event_target:es_theocracy_province
					female = yes
				}
			}
		}
	}
	
	after = {
		clr_country_flag = in_theocracy_heir_selection
	}
}
