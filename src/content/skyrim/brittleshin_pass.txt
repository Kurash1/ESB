new localisation south_brittleshin_exp_tooltip = "Unlocks a Great Project §YBrittleshin Pass§! in §YSouth Brittleshin§! province"
new expedition south_brittleshin = {
	picture = brittleshin_eventPicture
	length = 1
	on_complete = {
		set_province_flag = brittleshin_pass
		custom_tooltip = south_brittleshin_exp_tooltip
	}
}
new localisation brittleshin_pass_tooltip = "Connects §YSouth Brittleshin§! to §YNorth Brittleshin§!"
new great_project brittleshin_pass = {
	name = "Brittleshin Pass"
	start = south_brittleshin
	type = canal
	date = 3000.1.1
	starting_tier = 3
	time = 60
	can_be_moved = no
	build_cost = 500
	build_trigger = {
		has_province_flag = brittleshin_pass
	}
	on_built = {
		add_canal = brittleshin_pass
	}
	on_destroyed = {
		remove_canal = brittleshin_pass
	}
	tier_0 = {
		on_upgraded = {
			custom_tooltip = brittleshin_pass_tooltip
		}
	}
	tier_1 = {
		on_upgraded = {
			custom_tooltip = brittleshin_pass_tooltip
		}
	}
	tier_2 = {
		on_upgraded = {
			custom_tooltip = brittleshin_pass_tooltip
		}
	}
	tier_3 = {
		on_upgraded = {
			custom_tooltip = brittleshin_pass_tooltip
		}
	}
}
new adjacency brittleshin_pass = {
	from = north_brittleshin
	to = south_brittleshin
	through = south_brittleshin
	type = canal
}
