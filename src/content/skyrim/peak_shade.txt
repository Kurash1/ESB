new localisation peak_shade_exp_tooltip = "Unlocks a Great Project §YSouth Shriekwind Bastion§! in §YPeak Shaded§! province"
new expedition peak_shade = {
	picture = south_shriekwind_eventPicture
	length = 1
	on_complete = {
		set_province_flag = south_shriekwind
		custom_tooltip = peak_shade_exp_tooltip
	}
}
new localisation south_shriekwind_tooltip = "Connects §YSouth Shriekwind§! to §YShriekwind Bastion§!"
new great_project south_shriekwind = {
	name = "South Shriekwind Bastion"
	start = peak_shade
	type = canal
	date = 3000.1.1
	starting_tier = 3
	time = 60
	can_be_moved = no
	build_cost = 500
	build_trigger = {
		has_province_flag = south_shriekwind
	}
	on_built = {
		add_canal = south_shriekwind
	}
	on_destroyed = {
		remove_canal = south_shriekwind
	}
	tier_0 = {
		on_upgraded = {
			custom_tooltip = south_shriekwind_tooltip
		}
	}
	tier_1 = {
		on_upgraded = {
			custom_tooltip = south_shriekwind_tooltip
		}
	}
	tier_2 = {
		on_upgraded = {
			custom_tooltip = south_shriekwind_tooltip
		}
	}
	tier_3 = {
		on_upgraded = {
			custom_tooltip = south_shriekwind_tooltip
		}
	}
}
new adjacency south_shriekwind = {
	from = shriekwind_bastion
	to = peak_shade
	through = peak_shade
	type = canal
}
