new spell equilibrium = {
	name = "Equilibrium"
	type = country
	cost = 50
	on_cast = {
		new_custom_tooltip = "Lose §R50§! of your highest Power Type.
Gain §G25§! of the other two Power Types."
		hidden_effect = {
			&temp &= trigger_value:adm_power
			&temp_2 &= trigger_value:dip_power
			if [&temp >= temp_2] {
				&temp_2 &= trigger_value:mil_power
				if [&temp >= temp_2] {
					add_adm_power = -50
					add_dip_power = 25
					add_mil_power = 25
				}
				else {
					add_adm_power = 25
					add_dip_power = 25
					add_mil_power = -50
				}
			}
			else {
				&temp &= trigger_value:mil_power
				if [&temp_2 >= temp] {
					add_adm_power = 25
					add_dip_power = -50
					add_mil_power = 25
				}
				else {
					add_adm_power = 25
					add_dip_power = 25
					add_mil_power = -50
				}
			}
		}
	}
}
