			owns = falkreath
			provinces:falkreath = {
				development = 15
			}
		}
		effect = {
			provinces:falkreath = {
				quick_province_modifier = {
					name = "Falkreath City"
					modifier = {
						local_tax_modifier = 0.25
						local_production_efficiency = 0.25
						local_manpower_modifier = 0.25
						local_development_cost = -0.1
					}
				}
			}
		}
	}
}
on_actions:on_battle_won_country += {
	if [
		tag = skjoralmor
	] {
		change_variable = {
			which = total_casualties_caused
			which = land_units_killed
		}
	}
}
on_actions:on_battle_lost_country += {
	if [
		tag = skjoralmor
	] {
		change_variable = {
			which = total_casualties_caused
			which = land_units_killed
		}
	}
}
mission_trees:skjoralmor += {
	heroes_graveyard = {
		name = "Heroes' Graveyard"
		required = falkreath
		x = 6
		y = 3
		trigger = {
			defineloc total_casualties_caused = "Total Casualties Caused"
			check_variable = {
				which = total_casualties_caused
				value = 250000
			}
		}
		effect = {
			quick_country_modifier = {
				name = "Heroes' Graveyard"
				modifier = {
					morale_damage_received = -0.05
					reserves_organisation = 0.10
				}
			}
		}
	}
	high_hall = {
		name = "High Hall"
		required = heroes_graveyard
		x = 5
		y = 4
		trigger = {
			owns = falkreath
			government_rank = 4
			monthly_adm = 8
			monthly_dip = 8
			monthly_mil = 8
		}
		effect = {
			provinces:black_mountain = {
				move_great_project = {
					type = hakkvilds_high_hall
					province = falkreath
					instant = yes
				}
			}
		}
	}
}
