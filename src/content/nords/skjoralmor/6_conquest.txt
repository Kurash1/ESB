new effect skjoralmor_siege_effect = {
	args = {
		name = string
		province = province
		country = country
	}
	transpile = {
		new_custom_tooltip = "The Orc Stronghold of §Y{args:name}§! is destroyed
§Y[This.GetName]§! gains a permanent claim on all provinces owned by §Y{args:name}§!"
		hidden_effect = {
			args:province = {
				remove_orc_stronghold = yes
			}
			args:country = {
				every_owned_province = {
					add_permanent_claim = ROOT
				}
			}
		}
	}
}
