		name = "Skjoralmor Deal"
		x = 3
		y = 0
		trigger = {
			if [has_global_flag = skjoralmor_deal] {
				has_global_flag = skjoralmor_deal
			}
			else {
				treasury = 250
				manpower = 15
			}
		}
		effect = {
			if [NOT = { has_global_flag = skjoralmor_deal }] {

					add_treasury = 250
					add_manpower = 15

						add_permanent_claim = skjoralmor
					}
				}
				set_global_flag = skjoralmor_deal
			}
		}
	}
