new mission_series fallowfire_2 = { 
	slot = 2
	ai = yes
	potential = {
		tag = fallowfire
	}
	missions = {
		fallowfire_prepare_our_men_for_war = {
			name = "Prepare our men for war"
			desc = ""
			position = 0
			icon = armory_mission
			trigger = {
				army_size_percentage = 1.0
			}
			effect = {
				add_country_modifier = {
					name = inspired_men
					duration = (365*15)
				}
			}
		}
		fallowfire_clan_northwind = {
			name = "Clan Northwind"
			desc = ""
			position = 2
			icon = mission_rb_control_electors
			required = {
				fallowfire_finishing_our_rivals
			}
			trigger = {
				OR = {	
					countries:north-wind = {
						OR = {
							has_opinion = {
								who = fallowfire
								value = 190
							}
							is_vassal = yes
						}
					}
					provinces:mistwatch, provinces:mist_road = {
						type = all
						owns_or_non_sovereign_subject_of = fallowfire
					}
				}
			} 
			effect = {
				if [countries:north-wind = {
						has_opinion = {
							who = fallowfire
							value = 190
						}
					}
				] {
					countries:fallowfire = {
						vassalize = north-wind
					}
				}
				else {
					areas:ansilvund, areas:northwind, areas:mistwatch = {
						quick_province_modifier = {
							name = "Absolute control"
							duration = (30*365)
							modifier = {
								local_autonomy = -0.10
								local_monthly_devastation = -0.1
							}
						}
					}
				}
					if [countries:fallowfire = {mission_completed = }] {
						country_event = { 
							id = fallowfire.1
							days = 30
						}
						country_event = {
							id = fallowfire.8
							days = 0
						}
					}
			} 
		}
	}
}
