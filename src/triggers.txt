new trigger has_advisor_level_three_of = {
	args = string
	transpile = {
		custom_trigger_tooltip = {
			tooltip = `has_level_three_{args}_advisors`
			OR = {
				foreach $advisor_type in advisor_types [
					str:contains = {
						string = $advisor_type:monarch_power
						value = args
					}
				] {
					$advisor_type = 3
				}
			}
		}
	}
}
