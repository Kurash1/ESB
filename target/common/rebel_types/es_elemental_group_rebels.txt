sload_faith_rebels = {

	color = { 179 51 51 }
	area = nation 					# Sticks to provinces of their nation
	government = theocracy 			# Desires Theocracy
	religion = sload_faith
	defection = none				# Will not defect
	defect_delay = 120				# Held provinces will auto-defect after 18 months, regardless of accept/enforce demands.
	independence = none				# Does not strive for independence
	unit_transfer = no
	gfx_type = culture_province
	will_relocate = yes
	
	resilient = no
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.0
	infantry = 0.7
	cavalry = 0.3
	
	morale = 1.1

	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0.5
			is_overseas = yes
		}
		modifier = {
			factor = 0.0
			owner = { religion = ROOT }
		}
		modifier = {
			factor = 0.0
			NOT = { religion = sload_faith }
		}
		modifier = {
			factor = 0.001
			tolerance_to_this = 5
		}
		modifier = {
			factor = 0.5
			tolerance_to_this = 2
		}
		modifier = {
			factor = 5.0
			owner = { has_country_flag = religious_turmoil }
		}		
		modifier = {
			factor = 2000
			has_missionary = yes
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = 1 }
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = -1 }
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = -2 }
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.5
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 0.0
			religion = sload_faith
			owner = { religion = sload_faith }
		}
		modifier = {
			factor = 2.0
			NOT = { religion = sload_faith }
			any_neighbor_province = {
	 			religion = sload_faith
	 			NOT = { has_province_modifier = "forced_conversion" }
	 		}
	 	}
		modifier = {
			factor = 1.5
			unrest = 2.5
		}
		modifier = {
			factor = 1.5
			unrest = 5
		}
		modifier = {
			factor = 1.5
			unrest = 7.5
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		NOT = {
			AND = {
				always = no
				is_capital = yes
			}
		}
 	}
	siege_won_effect = { custom_tooltip = religious_rebels_seige_effect_tooltip owner = { change_estate_land_share = { estate = estate_priests share = 10.0 province = ROOT } }
		if = {
			limit = {
				religion = sload_faith
			}
			add_permanent_province_modifier = {
				name = "religious_violence"
				duration = 730
			}
		}
		if = {
			limit = {
				NOT = { religion = sload_faith }
			}
			change_religion = sload_faith
			add_permanent_province_modifier = {
				name = "forced_conversion"
				duration = 730
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		OR = {
			religion_group = elemental_group
			dominant_religion = sload_faith
		}
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "sload_faith_rebels_demand"
	
	# Country Scope
	demands_enforced_effect = { hidden_effect = { add_country_modifier = { name = just_lost_to_rebels duration = 10 hidden = yes } change_estate_land_share = { estate = estate_priests share = 10.0 }
		if = {
			limit = {
				NOT = { religion_group = elemental_group }
			}
			add_prestige = -50
			every_owned_province = {
				limit = {
					has_rebel_faction = REB
				}
				add_local_autonomy = 25
			}
		}
		if = {
			limit = {
				always = no
			}
			add_stability = -1
		}
		if = {
			limit = {
				religion = sload_faith
			}
			add_country_modifier = {
				name = "religious_intolerance"
				duration = 3650
			}
		}
		if = {
			limit = {
				NOT = { religion = sload_faith }
				dominant_religion = sload_faith
			}
			change_religion = sload_faith
			force_converted = yes
		}
		if = {
			limit = {
				NOT = { religion = sload_faith }
				NOT = { dominant_religion = sload_faith }
				religion_group = elemental_group
			}
			add_country_modifier = {
				name = "heretic_tolerance"
				duration = 3650
			}
		}
	}
	}
}
