new effect country_event = {
	args = {
		id = country_event
		days = int?
		random = int?
		tooltip = string?
	}
	transpile = {
		`country_event =` {
			`id = {args:id}`
			when [exists = args:days] `days = {args:days}`
			when [exists = args:random] `random = {args:random}`
			when [exists = args:tooltip] `tooltip = {args:tooltip}`
		}
	}
}
