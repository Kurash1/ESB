new effect country_event = {
	args = {
		id = country_event
		years = int?
		random = int?
		tooltip = string?
	}
	transpile = {
		`country_event =` {
			when [exists = years] `days = {(args:years*365)}`
			when [exists = random] `random = {args:random}`
			when [exists = tooltip] `tooltip = {args:tooltip}`
		}
	}
}
