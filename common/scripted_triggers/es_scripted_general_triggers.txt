# AI War Declaration

can_declare_scenario_war = {
	custom_trigger_tooltip = {
        tooltip = can_declare_scenario_war_tooltip
		OR = {
			ai = no
			AND = {
				is_at_war = no
				stability = 0
				has_any_disaster = no
				NOT = { war_exhaustion = 10 }
				manpower_percentage = 0.25
			}
		}
	}
}
